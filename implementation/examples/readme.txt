To start the examples from the build-directory do:

env VSOMEIP_CONFIGURATION_FILE=../config/vsomeip.xml VSOMEIP_APPLICATION_NAME=client-sample ./client-sample
env VSOMEIP_CONFIGURATION_FILE=../config/vsomeip.xml VSOMEIP_APPLICATION_NAME=service-sample ./service-sample
env VSOMEIP_CONFIGURATION_FILE=../config/vsomeip.xml VSOMEIP_APPLICATION_NAME=other-client-sample ./client-sample
